1function [Psi , Theta, Phi ]   = fcn(r1, r2 , r3, r4 ,X, Y, Psi , U , V, Phi,    P , The, Q , R, Z, time)
 2coder.extrinsic('sim');
 3coder.extrinsic('timeseries');
 4global previousTime;
 5global sim_error;
 6global sim_iteration;
 7global results;
 8sim_time = 2;
 9
10if(previousTime+1 > time)
11    previousTime = int16(time);
12    n = 3;
13
14    result = zeros(1,n);
15    Phi_blend = zeros(1,n);
16    Theta_blend = zeros(1,n);
17    Psi_blend = zeros(1,n);
18
19    sim_state.X = X;
20    sim_state.Y = Y;
21    sim_state.Psi = Psi;
22    sim_state.U = U;
23    sim_state.V = V;
24    sim_state.Phi = Phi;
25    sim_state.P = P;
26    sim_state.The = The;
27    sim_state.Q = Q;
28    sim_state.R = R;
29    sim_state.Z = Z;
30    
31    for i = 1:n
32        %initlize
33        sim_error = 0;
34        
35        sim_state.X = X;
36        sim_state.Y = Y;
37        sim_state.Psi = Psi;
38        sim_state.U = U;
39        sim_state.V = V;
40        sim_state.Phi = Phi;
41        sim_state.P = P;
42        sim_state.The = The;
43        sim_state.Q = Q;
44        sim_state.R = R;
45        sim_state.Z = Z;
46        
47        error1 = r1;
48        error2 = r2;
49        error3 = r3; 
50        error4 = r4;
51        sim_iteration = 0;
52        
53        %generate sim path
54        sim_path.x = 0;
55        sim_path.y = 0;
56        sim_path.z = 0;
57        sim_path.psi = 0;
58        
59        X_goal = X+sim_time;
60        X_t = [ X, X_goal]; % meters
61        Y_t = [ Y, 0]; % meters
62        Z_t = [ Z, 3]; % meters
63        t = [ 0, sim_time]; % seconds
64        Psi_timeseries = [Psi, 0]; % radians
65        sim_path.x = timeseries(X_t,t);
66        sim_path.y = timeseries(Y_t,t);
67        sim_path.z = timeseries(Z_t,t);
68        sim_path.psi = timeseries(Psi_timeseries,t);
69       
70        %gen random blends
71        Phi_sim_blend = (rand()*100)/100;
72        Theta_sim_blend = (rand()*100)/100;
73        Psi_sim_blend = (rand()*100)/100;
74
75        %simulate for N steps
76
77        sim('Partial_Quadcopter_simulation');    
78
79        %record sim result
80      
81        result(n) = sim_error;
82        Phi_blend(n) = Phi_sim_blend;
83        Theta_blend(n) = Theta_sim_blend;
84        Psi_blend(n) = Psi_sim_blend ;
85
86
87
88    end
89    
90    %update Q matrices
91end
92
93%output next step for 
94[ val , index ] = min(result);
95Psi = Phi_blend(index);
96Theta = Theta_blend(index);
97Phi = Phi_blend(index);
